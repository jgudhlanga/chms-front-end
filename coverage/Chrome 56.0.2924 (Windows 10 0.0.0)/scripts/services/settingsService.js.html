'use strict';
(function(){
 
    var SettingsService = function($http)
    {
        //1 get the all modules
        var listModules = function(){
            return $http.get("http://www.jdev-systems/api/v1/modules").then(
                function(response)
                {
                    return response.data;
                },
                function(errorMsg){
                    return errorMsg;
                }
            );
        };
 
        /*get a module */
        var getModule = function(moduleId){
            return $http.get('http://www.jdev-systems/api/v1/modules/'+moduleId+'').then(
                function(response)
                {
                    return response.data;
                },
                function(errorMsg){
                    return errorMsg;
                }
            );
        };
 
        /* save module */
        var config = {
            headers: {
                'content-type': 'application/json',
                'Authorization': ''
            }
        }
        var saveModule = function(formData)
        {
            return $http.post('http://www.jdev-systems/api/v1/modules',formData, config)
               .then(function(response){
                return response.data;
            },
             function(reason){
                 return reason;
            });
        };
        /* update module */
        var updateModule = function(formData)
        {
            return $http.put('http://www.jdev-systems/api/v1/modules/'+formData.id, formData, config)
                .then(function(response){
                    return response.data;
                },
                 function(reason){
                     return reason;
                });
        };
        /* activate - deactivate  module */
        var activateModule = function(moduleId, active)
        {
            return $http.patch('http://www.jdev-systems/api/v1/modules/'+moduleId+'/'+active, config)
                .then(function(response){
                    return response.data;
                },
                 function(reason){
                     return reason;
            });
        };
 
        /* activate - deactivate  module */
        var orderModule = function(moduleId, direction)
        {
            return $http.patch('http://www.jdev-systems/api/v1/modules/order/'+moduleId+'/'+direction, config)
                .then(function(response){
                    return response.data;
                },
                 function(reason){
                     return reason;
            });
        };
 
        /* delete module */
        var deleteModule = function(module)
        {
            return $http.delete('http://www.jdev-systems/api/v1/modules/'+module, config)
                .then(function(response){
                return response.data;
            },
             function(reason){
                 return reason;
            });
        };
        //menu icons object
        var jdevMenuIcons = {
            'fa fa-gears': 'fa fa-gears', 'fa fa-database': 'fa fa-database', 'fa fa-calendar': 'fa fa-calendar',
            'fa fa-dashboard': 'fa fa-dashboard','fa fa-question-circle': 'fa fa-question-circle', 'fa fa-home': 'fa fa-home',
            'fa fa-group': 'fa fa-group', 'fa fa-folder': 'fa fa-folder', 'fa fa-bar-chart': 'fa fa-bar-chart',
            'fa fa-wrench': 'fa fa-wrench'
        };
 
        //Return the object of action in this service
        return {
            jdevMenuIcons: jdevMenuIcons,
            listModules: listModules,
            getModule: getModule,
            updateModule: updateModule,
            saveModule: saveModule,
            activateModule: activateModule,
            orderModule: orderModule,
            deleteModule: deleteModule
        };
    };
 
    /* Register the SettingsService with the module */
 
    angular.module('jdevAngularApp').factory('SettingsService',SettingsService);
}());
