'use strict';
 
/**
 * @ngdoc overview
 * @name jdevAngularApp
 * @description
 * # jdevAngularApp
 *
 * Main module of the application.
 */
angular
  .module('jdevAngularApp', [
    'ngAnimate',
    'ngAria',
    'ngCookies',
    'ngMessages',
    'ngResource',
    'ngRoute',
    'ngSanitize',
    'ngTouch'
  ])
  .value('version', '0.0.1')
  .config(function ($routeProvider, $locationProvider, $compileProvider) {
    $routeProvider
    .when('/', {
        templateUrl: 'views/main.html'
    })
    /* Admin Routes  */
    .when('/admin', {
        templateUrl: 'views/admin/index.html',
        controller: 'AdminCtrl',
        controllerAs: 'AC'
    })
    /* Backup Routes  */
    .when('/backup', {
        templateUrl: 'views/backup/index.html',
        controller: 'BackupCtrl',
        controllerAs: 'BC'
    })
    /* Calendar Routes  */
    .when('/calendar', {
        templateUrl: 'views/calendar/index.html',
        controller: 'CalendarCtrl',
        controllerAs: 'CC'
    })
    /* Dashboard Routes  */
    .when('/dashboard', {
        templateUrl: 'views/dashboard/index.html',
        controller: 'DashboardCtrl',
        controllerAs: 'DAC'
    })
    /* Departments Routes  */
    .when('/departments', {
        templateUrl: 'views/departments/index.html',
        controller: 'DepartmentsCtrl',
        controllerAs: 'DC'
    })
    /* Families Routes  */
    .when('/families/index', {
        templateUrl: 'views/families/index.html',
        controller: 'FamiliesCtrl',
        controllerAs: 'FC'
    })
    /* Groups Routes  */
    .when('/groups/index', {
        templateUrl: 'views/groups/index.html',
        controller: 'GroupsCtrl',
        controllerAs: 'GC'
    })
    /* Help Routes  */
    .when('/help', {
        templateUrl: 'views/help/index.html',
        controller: 'HelpCtrl',
        controllerAs: 'HC'
    })
    /* Library Routes  */
    .when('/library', {
        templateUrl: 'views/library/index.html',
        controller: 'LibraryCtrl',
        controllerAs: 'LC'
    })
    /* Members Routes  */
    .when('/members/index', {
        templateUrl: 'views/members/index.html',
        controller: 'MembersCtrl',
        controllerAs: 'MC'
    })
    /* Ministries Routes  */
    .when('/ministries/index', {
        templateUrl: 'views/ministries/index.html',
        controller: 'MinistriesCtrl',
        controllerAs: 'MC'
    })
    /* Projects Routes  */
    .when('/projects/index', {
        templateUrl: 'views/projects/index.html',
        controller: 'ProjectsCtrl',
        controllerAs: 'PC'
    })
    /* Reports Routes  */
    .when('/reports', {
        templateUrl: 'views/reports/index.html',
        controller: 'ReportsCtrl',
        controllerAs: 'RC'
    })
    /* Settings Routes */
    .when('/settings', {
        templateUrl: 'views/settings/settings.html'
    })
    .when('/addeditmodule/:moduleId', {
        templateUrl: 'views/settings/addeditmodule.html',
        controller: 'SettingsCtrl',
        controllerAs: 'SC'
     })
    .when('/addeditmodule', {
        templateUrl: 'views/settings/addeditmodule.html',
        controller: 'SettingsCtrl',
        controllerAs: 'SC'
     })
    .when('/modulepages/:moduleId', {
        templateUrl: 'views/settings/modulepages.html',
        controller: 'SettingsCtrl',
        controllerAs: 'SC'
    })
    /* Visitors Routes  */
    .when('/visitors/index', {
        templateUrl: 'views/visitors/index.html',
        controller: 'VisitorsCtrl',
        controllerAs: 'VC'
    })
    /* Zones Routes  */
    .when('/zones/index', {
        templateUrl: 'views/zones/index.html',
        controller: 'ZonesCtrl',
        controllerAs: 'ZC'
    })
    .otherwise({
        redirectTo: '/'
    });
    $compileProvider.debugInfoEnabled(false);
    $locationProvider.html5Mode(true);
  });
